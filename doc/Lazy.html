def method_missing(*args, &block)
  unless @done
    if !@args && !@block
      @args, @block = args, @block
      return self
    end

    @res = @args ?  @obj.send(*@args, &@block) : @block.call(@obj)
    @done = true
  end
  
  @res.send(*args, &block)
end
